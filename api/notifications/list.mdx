---
title: "List notifications"
description: "Get the authenticated user's in-app notifications."
api: "GET /notifications"
---

Returns a paginated list of notifications for the current user, sorted by creation date descending.

## Query parameters

<ParamField query="limit" type="integer" default="50">
  Number of notifications to return. Min `1`, max `100`.
</ParamField>

<ParamField query="offset" type="integer" default="0">
  Number of notifications to skip.
</ParamField>

## Response

<ResponseField name="notifications" type="array">
  <Expandable title="NotificationResponse">
    <ResponseField name="id" type="string">Notification UUID.</ResponseField>
    <ResponseField name="user_id" type="string">Recipient user UUID.</ResponseField>
    <ResponseField name="channel" type="string">Delivery channel: `in_app`, `email`.</ResponseField>
    <ResponseField name="subject" type="string">Notification title.</ResponseField>
    <ResponseField name="body" type="string">Notification content.</ResponseField>
    <ResponseField name="status" type="string">Delivery status: `pending`, `sent`, `delivered`, `failed`.</ResponseField>
    <ResponseField name="priority" type="string">Priority: `low`, `normal`, `high`.</ResponseField>
    <ResponseField name="created_at" type="string">ISO 8601 creation timestamp.</ResponseField>
    <ResponseField name="sent_at" type="string">When the notification was sent. `null` if not yet sent.</ResponseField>
    <ResponseField name="delivered_at" type="string">When delivery was confirmed. `null` if not confirmed.</ResponseField>
    <ResponseField name="read_at" type="string">When the user read the notification. `null` if unread.</ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="total" type="integer">
  Total number of notifications.
</ResponseField>

<ResponseField name="unread_count" type="integer">
  Number of unread notifications.
</ResponseField>
